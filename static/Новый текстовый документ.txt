<script>
    let monthlyChart = null;
    let categoryChart = null;
    
    async function uploadFile() {
        const file = document.getElementById('fileInput').files[0];
        if (!file) {
            showStatus('Выберите файл', 'error');
            return;
        }
        
        const formData = new FormData();
        formData.append('file', file);
        
        showStatus('Загрузка и анализ...', '');
        
        try {
            const response = await fetch('/excel/upload', {
                method: 'POST',
                body: formData
            });
            
            const data = await response.json();
            
            if (response.ok && data.status === 'ok') {
                showStatus('✅ Анализ завершён успешно!', 'success');
                displayResults(data);
            } else {
                showStatus('❌ Ошибка: ' + JSON.stringify(data), 'error');
            }
        } catch(e) {
            showStatus('❌ Ошибка: ' + e.message, 'error');
        }
    }
    
    function showStatus(msg, type) {
        const statusDiv = document.getElementById('status');
        statusDiv.textContent = msg;
        statusDiv.className = 'status ' + type;
    }
    
    function displayResults(data) {
        document.getElementById('results').style.display = 'block';
        
        // Статистика
        const statsHtml = `
            <div class="stat-card"><h3>Общие расходы</h3><div class="value">${data.total.toLocaleString()} ₽</div></div>
            <div class="stat-card"><h3>Период</h3><div class="value">${data.months.length} месяцев</div></div>
            <div class="stat-card"><h3>Категории</h3><div class="value">${data.table_data.length}</div></div>
            <div class="stat-card"><h3>Ср. в месяц</h3><div class="value">${Math.round(data.total / data.months.length).toLocaleString()} ₽</div></div>
        `;
        document.getElementById('stats').innerHTML = statsHtml;
        
        // Таблица
        const tbody = document.getElementById('tableBody');
        tbody.innerHTML = '';
        data.table_data.forEach(row => {
            tbody.innerHTML += `<tr><td>${row.category}</td><td>${row.amount.toLocaleString()}</td><td>${row.percent}%</td></tr>`;
        });
        
        // График по месяцам
        if (monthlyChart) monthlyChart.destroy();
        monthlyChart = new Chart(document.getElementById('monthlyChart'), {
            type: 'line',
            data: {
                labels: data.months,
                datasets: [{ label: 'Расходы (₽)', data: data.totals_by_month, borderColor: '#c47a5e', backgroundColor: '#c47a5e20', fill: true, tension: 0.3 }]
            },
            options: { responsive: true, plugins: { legend: { position: 'top' } } }
        });
        
        // Круговая диаграмма
        if (categoryChart) categoryChart.destroy();
        categoryChart = new Chart(document.getElementById('categoryChart'), {
            type: 'pie',
            data: {
                labels: data.chart_data.map(d => d.category),
                datasets: [{ data: data.chart_data.map(d => d.amount), backgroundColor: ['#c47a5e','#e7b183','#a5c882','#8b7a66','#c4a27a','#b68b6e','#d4a373','#9c6b4a'] }]
            },
            options: { responsive: true, plugins: { legend: { position: 'bottom' } } }
        });
        
        // Рекомендации
        document.getElementById('recommendations').innerHTML = '<h3>💡 Рекомендации AI</h3><ul>' + data.recommendations.map(r => `<li>${r}</li>`).join('') + '</ul>';
    }
</script>
